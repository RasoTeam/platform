def self.search(search,order,state=nil)
  order ||= ""
  if search
    companies = where('name LIKE ? OR slug LIKE ?', '%'+search+'%', '%'+search+'%').order("slug "+order)
  else
    companies = order("slug "+order)
  end
  if !state.blank?
    companies = companies.where('state = ?', Integer(state))
  end
  return companies
end
